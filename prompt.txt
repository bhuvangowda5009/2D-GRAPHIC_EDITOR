For Day 1, use this complete program:

#include <stdio.h>

#define ROWS 25
#define COLS 60

char picture[ROWS][COLS];

/* Fill the canvas with '_' */
void clearPicture()
{
    int i, j;

    for(i = 0; i < ROWS; i++)
    {
        for(j = 0; j < COLS; j++)
        {
            picture[i][j] = '_';
        }
    }
}

/* Display the canvas */
void displayPicture()
{
    int i, j;

    for(i = 0; i < ROWS; i++)
    {
        for(j = 0; j < COLS; j++)
        {
            printf("%c", picture[i][j]);
        }
        printf("\n");
    }
}

int main()
{
    printf("DAY 1 : BLANK 2D CANVAS\n\n");

    clearPicture();
    displayPicture();

    return 0;
}

Creates a 2D array:
char picture[25][60];
